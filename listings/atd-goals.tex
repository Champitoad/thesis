(* -------------------------------------------------------------------- *)
(** Goals *)

(* Unique identifier *)
type uid = string

(* Hypothesis *)
type hyp = {
  h_id : uid;
  h_form : form;
}

(* Goal *)
type goal = {
  g_env : env;
  g_hyps : hyp list;
  g_concl : form;
}

type goals = goal list

(* Abstract goal, without the signature *)
type agoal = {
  a_vars : varenv;
  a_hyps : hyp list;
  a_concl : form;
}

\end{minted}
